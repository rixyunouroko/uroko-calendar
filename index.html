function searchBirth() { const y = parseInt(document.getElementById("birthYear").value, 10); const m = parseInt(document.getElementById("birthMonth").value, 10); const d = parseInt(document.getElementById("birthDay").value, 10); const box = document.getElementById("birthResult"); if (!y || !m || !d) { box.textContent = "年・月・日を全部入れてから検索してな。"; return; } const date = new Date(y, m - 1, d); if (isNaN(date.getTime())) { box.textContent = "日付がおかしいかも。もう一度確認してみて。"; return; } const { tone, sign } = getMayanFor(date); const desc = sign.birthDescription ? sign.birthDescription : `◎ 紋章のテーマ\n${sign.frontKeyword}\n\n（誕生日用の説明はまだ未設定やで）`; box.innerHTML = `【${formatDateJa(date)} 生まれのメモ】\n\n` + `${desc}\n\n` + `――――――\n` + `◎ あなたの音：音${tone.num}（${tone.keyword}）\n\n` + `【音${tone.num}の流れ】${tone.front}\n` + `【つまずきポイント】${tone.back}\n\n` + `※ この結果はウロコ版のツォルキン計算やで。他のサイトと違うこともあるけど、\n` + `　このアプリ内では全部この暦で揃えてあると思って使ってな。`; }
